// ── WHEEL SPEED SENSOR (Hall Effect) — BARU v2.4 ─────────────────
// Sesuaikan dengan konfigurasi kendaraan Anda:
#define WHEEL_CIRC_M       1.92f   // Keliling ban dalam meter
                                   // Contoh: ban 195/65 R15 → ~1.92m
                                   // Rumus: π × diameter_total_m
                                   // diameter_total = rim_m + 2×sidewall_m
                                   // Untuk 195/65R15: (0.381 + 2×0.1267)×π ≈ 1.918m
#define MAGNETS_PER_REV    4       // Jumlah magnet di piringan roda
                                   // 1 magnet = resolusi rendah, update lambat
                                   // 4 magnet = resolusi tinggi, sangat responsif
                                   // (rekomendasi untuk kecepatan rendah / presisi tinggi)
#define WSS_CALC_INTERVAL  50      // ms — kalkulasi speed setiap 50ms (20Hz)
#define WSS_TIMEOUT_MS     500     // ms — kecepatan → 0 jika tidak ada pulse
#define WSS_SPEED_ALPHA    0.35f   // Low-pass filter coefficient
                                   // 0.0=sangat halus (lambat), 1.0=langsung (noise)
                                   // 0.35 = responsif tapi anti-getaran
#define WSS_MIN_PULSE_US   800     // Minimum interval antar pulse (µs)
                                   // Proteksi dari bounce/noise elektrik
                                   // 800µs ≈ kecepatan max ~864 km/h (4 magnet ban std)

void IRAM_ATTR rpm_isr() {

}

// ═══════════════════════════════════════════════════════════════════
// ISR — WHEEL SPEED SENSOR (Hall Effect Roda) — BARU v2.4
// ═══════════════════════════════════════════════════════════════════
// Dipanggil setiap magnet melewati sensor (FALLING edge, aktif LOW)
// Disimpan di IRAM untuk latensi minimal — kecepatan tinggi butuh ISR cepat

void IRAM_ATTR wss_isr() {
  uint32_t now_us = micros();
  uint32_t interval = now_us - wss_last_us;

  // Hardware debounce: abaikan pulse terlalu cepat (noise / bounce)
  if(interval < WSS_MIN_PULSE_US) return;

  wss_pulses++;
  wss_last_us    = now_us;
  wss_pulse_valid = true;
}

// ═══════════════════════════════════════════════════════════════════
// SENSOR READERS

float read_volt() {
  uint16_t adc = analogRead(VOLT_PIN);
  float v = adc * 3.3f / 4095.0f;
  return constrain(v * 5.0f, 0.0f, 16.0f);
}

int read_gear() {
  uint32_t now = millis();
  int active_count = 0;
  int active_idx   = -1;
  for(int i = 0; i < 6; i++) {
    bool raw = (digitalRead(GEAR_PINS[i]) == LOW);
    if(raw != gear_state[i]) {
      if((now - gear_debounce_ms[i]) >= GEAR_DEBOUNCE_MS) {
        gear_state[i]       = raw;
        gear_debounce_ms[i] = now;
      }
    } else {
      gear_debounce_ms[i] = now;
    }
    if(gear_state[i]) { active_count++; active_idx = i; }
  }
  if(active_count > 1) return gear;
  if(active_count == 0) {
    if(!gear_neutral_pending) {
      gear_neutral_pending = true;
      gear_neutral_timer   = now;
    }
    if((now - gear_neutral_timer) >= GEAR_NEUTRAL_DELAY_MS) {
      gear_neutral_pending = false;
      return -1;
    }
    return gear;
  }
  gear_neutral_pending = false;
  // index 0=R→return 0, index 1=G1→return 1, dst
  return active_idx;
}

// ─── WHEEL SPEED SENSOR — UPDATE KECEPATAN — BARU v2.4 ───────────
//
// Kalkulasi kecepatan dari Hall Effect roda setiap WSS_CALC_INTERVAL ms.
// Metode: hitung pulse dalam interval → konversi ke km/h via keliling ban.
//
// Formula:
//   RPS (rev/s) = (pulses / interval_s) / MAGNETS_PER_REV
//   v (m/s)     = RPS × WHEEL_CIRC_M
//   v (km/h)    = v (m/s) × 3.6
//
// Low-pass filter mencegah nilai bergetar akibat toleransi magnet/sensor:
//   speed_filtered = (1-α)×speed_filtered + α×speed_raw
//
// Timeout: jika tidak ada pulse selama WSS_TIMEOUT_MS → kecepatan = 0
// (kendaraan berhenti, tidak ada edge yang trigger ISR)

static unsigned long wss_last_count   = 0;
static uint32_t      wss_last_calc_ms = 0;

void update_wss_speed() {
  uint32_t now = millis();
  if(now - wss_last_calc_ms < WSS_CALC_INTERVAL) return;

  float dt_s = (now - wss_last_calc_ms) / 1000.0f;
  wss_last_calc_ms = now;

  // Ambil jumlah pulse dalam interval ini (atomic read)
  noInterrupts();
  unsigned long current_pulses = wss_pulses;
  interrupts();

  unsigned long delta = current_pulses - wss_last_count;
  wss_last_count = current_pulses;

  // Cek timeout: tidak ada pulse → berhenti
  uint32_t last_us = wss_last_us;   // baca volatile sekali
  uint32_t elapsed_since_pulse = (micros() - last_us) / 1000; // konversi ke ms

  if(elapsed_since_pulse > WSS_TIMEOUT_MS) {
    // Kendaraan berhenti — langsung 0 tanpa filter (responsive stop)
    speed_raw      = 0.0f;
    speed_filtered = 0.0f;
    wss_ok  = false;
    speed   = 0;
    return;
  }

  if(delta > 0) {
    // Hitung kecepatan dari pulse
    float rps       = ((float)delta / dt_s) / (float)MAGNETS_PER_REV;
    float v_ms      = rps * WHEEL_CIRC_M;
    speed_raw       = v_ms * 3.6f;
    speed_raw       = constrain(speed_raw, 0.0f, 350.0f); // cap fisik

    // Low-pass filter: blending smooth
    speed_filtered  = (1.0f - WSS_SPEED_ALPHA) * speed_filtered
                    + WSS_SPEED_ALPHA * speed_raw;

    wss_ok = true;
  }
  // Jika delta==0 tapi belum timeout: pertahankan speed_filtered agar
  // tidak langsung drop ke 0 (normal di kecepatan sangat rendah)

  speed = (int)(speed_filtered + 0.5f); // round ke int untuk display

  // Accumulate trip odometer: v_km/h × dt_h = km
  if(speed_filtered > 1.0f) {
    double km_inc = (speed_filtered / 3600.0) * dt_s;
    wss_trip_km  += km_inc;
    wss_total_km += km_inc;
    trip_km       = wss_trip_km; // sync ke trip_km global (dipakai fuel calc)
  }
}
uint16_t fuel_adc_read_median() {
  uint16_t samples[FUEL_ADC_SAMPLES];
  for(int i = 0; i < FUEL_ADC_SAMPLES; i++) {
    samples[i] = analogRead(FUEL_ADC_PIN);
    delayMicroseconds(100);
  }
  // Bubble sort (small N)
  for(int i = 0; i < FUEL_ADC_SAMPLES - 1; i++) {
    for(int j = i + 1; j < FUEL_ADC_SAMPLES; j++) {
      if(samples[j] < samples[i]) {
        uint16_t t = samples[i]; samples[i] = samples[j]; samples[j] = t;
      }
    }
  }
  return (samples[FUEL_ADC_SAMPLES/2 - 1] + samples[FUEL_ADC_SAMPLES/2]) / 2;
}

float fuel_adc_to_pct(uint16_t adc) {
  if(fuel.adc_full == fuel.adc_empty) return 0.0f;
  float pct = (float)(adc - fuel.adc_empty) /
              (float)(fuel.adc_full - fuel.adc_empty) * 100.0f;
  return constrain(pct, 0.0f, 100.0f);
}

void update_fuel_sensor() {
  uint32_t now = millis();
  if(now - fuel.last_update_ms < FUEL_UPDATE_MS) return;
  fuel.last_update_ms = now;

  fuel.adc_raw   = fuel_adc_read_median();
  float raw_pct  = fuel_adc_to_pct(fuel.adc_raw);

  if(fabsf(raw_pct - fuel.level_pct_prev) >= FUEL_HYSTERESIS) {
    fuel.level_pct      = raw_pct;
    fuel.level_pct_prev = raw_pct;
  }
  fuel.level_pct_smooth = fuel.level_pct_smooth * 0.9f + fuel.level_pct * 0.1f;
  fuel.level_liter      = (fuel.level_pct_smooth / 100.0f) * FUEL_TANK_LITER;

  if(trip_km > 1.0 && fuel.trip_start_km >= 0) {
    double km_driven  = trip_km - fuel.trip_start_km;
    float  liter_used = fuel.trip_start_liter - fuel.level_liter;
    if(km_driven > 1.0 && liter_used > 0.0f) {
      float instant = constrain((liter_used / (float)km_driven) * 100.0f, 1.0f, 50.0f);
      fuel.cons_per_100km = fuel.cons_per_100km * (1.0f - FUEL_CONS_SMOOTH)
                           + instant * FUEL_CONS_SMOOTH;
    }
  } else {
    fuel.trip_start_liter = fuel.level_liter;
    fuel.trip_start_km    = (float)trip_km;
  }
  if(fuel.cons_per_100km < 0.1f) fuel.cons_per_100km = 10.0f;
  if(fuel.cons_per_100km > 0.1f) {
    fuel.range_km = (fuel.level_liter / fuel.cons_per_100km) * 100.0f;
  }

  bool new_warn = (fuel.level_pct_smooth < FUEL_LOW_WARN_PCT);
  bool new_crit = (fuel.level_pct_smooth < FUEL_CRIT_WARN_PCT);
  if(new_warn && !fuel.warn_shown) {
    fuel.warn_shown = true;
    notif_app   = "PERINGATAN BBM";
    notif_title = new_crit ? "BBM KRITIS!" : "BBM RENDAH";
    notif_msg   = "Sisa: " + String(fuel.level_liter, 1) + "L (~" +
                  String(fuel.range_km, 0) + " km)";
    notif_show = true;
    notif_ts   = millis();
  }
  if(!new_warn) fuel.warn_shown = false;
  fuel.warn_low  = new_warn;
  fuel.warn_crit = new_crit;

  // v2.5: Cek trigger navigasi Pertamina setiap update BBM
  check_pertamina_trigger();
}

// ─── GFORCE — Dari delta kecepatan Hall Effect (v2.4: bukan GPS) ──
void update_gforce(int cur_speed) {
  uint32_t now = millis();
  float dt = (now - gf_last_ms) / 1000.0f;
  if(dt < 0.1f) return;
  float delta_v = (cur_speed - last_speed_gf) / 3.6f;
  gforce        = constrain((delta_v / dt) / 9.81f, -2.0f, 2.0f);
  last_speed_gf = cur_speed;
  gf_last_ms    = now;
}

// ─── TRIP — Dihandle oleh update_wss_speed() di v2.4 ─────────────
// GPS trip (update_trip) digantikan oleh akumulasi jarak Hall Effect.
// Fungsi ini dipertahankan kosong untuk kompatibilitas GPS callback.
void update_trip(double cur_lat, double cur_lon) {
  // v2.4: trip_km diupdate oleh update_wss_speed() — lebih akurat & real-time
  // GPS lat/lon tetap disimpan untuk keperluan navigasi
  lat = cur_lat;
  lon = cur_lon;
}

// ─── ENGINE HOURS ─────────────────────────────────────────────────
void update_engine_hours() {
  bool on = (rpm > RPM_IDLE);
  if(on && !engine_running) {
    engine_start_ms = millis(); engine_running = true;
  } else if(!on && engine_running) {
    engine_total_s += (millis() - engine_start_ms) / 1000;
    engine_running  = false;
    nvs_save_engine_hours();
    nvs_save_odometer();   // v2.4: simpan odometer saat mesin mati
  }
}

// ─── EFFICIENCY ───────────────────────────────────────────────────
void update_efficiency() {
  if(rpm > RPM_IDLE && speed > 5) {
    float idx = ((float)speed / (float)rpm) / (80.0f / 2500.0f) * 100.0f;
    efficiency_idx = efficiency_idx * 0.85f + idx * 0.15f;
    efficiency_idx = constrain(efficiency_idx, 0.0f, 200.0f);
  }
}

// ─── AUTO DIM ─────────────────────────────────────────────────────
void update_auto_dim() {
  if(!clk_synced) return;
  bool dim = (clk_h >= DIM_HOUR_START || clk_h < DIM_HOUR_END);
  if(dim != is_night_mode) {
    is_night_mode = dim;
    gfx.setBrightness(dim ? DIM_BRIGHTNESS : BRIGHT_NORMAL);
    cur_brightness = dim ? DIM_BRIGHTNESS : BRIGHT_NORMAL;
  }
}

// ─── LAP TIMER ────────────────────────────────────────────────────
String format_lap_time(uint32_t ms) {
  int mins = ms / 60000;
  int secs = (ms % 60000) / 1000;
  int cent = (ms % 1000) / 10;
  char buf[16];
  snprintf(buf, sizeof(buf), "%d:%02d.%02d", mins, secs, cent);
  return String(buf);
}

void lap_timer_toggle() {
  if(!lap_running) {
    lap_start_ms = millis(); lap_running = true;
  } else {
    uint32_t t = millis() - lap_start_ms;
    lap_count++;
    if(lap_best_ms == 0 || t < lap_best_ms) lap_best_ms = t;
    lap_start_ms = millis();
  }
}

// ═══════════════════════════════════════════════════════════════════
// AUTO NAVIGASI PERTAMINA — BARU v2.5
// ═══════════════════════════════════════════════════════════════════
//
// Kirim request navigasi ke HP via Chronos BLE.
// Chronos meneruskan ke Google Maps dengan query + koordinat GPS.
//
// ChronosESP32 API yang digunakan:
//   watch.sendNotification() → untuk notif konfirmasi di HP
//   watch.setNavigation()    → set destination, HP buka Maps
//
// Format deep-link Google Maps yang diinject via Chronos:
//   "geo:<lat>,<lon>?q=SPBU+Pertamina"
//   Chronos akan parse lat/lon ini dan buka navigasi otomatis.
//
// Jika versi Chronos tidak support setNavigation(), fallback ke
// sendAlert() dengan URL Maps yang bisa diklik pengguna di HP.

void trigger_pertamina_navigation(bool is_critical) {
  // Guard: hanya jika semua kondisi terpenuhi
  if(!ble_connected) {
    Serial.println("[PERTAMINA] BLE tidak terhubung — navigasi dibatalkan");
    return;
  }
  if(!gps_ok || (lat == 0.0 && lon == 0.0)) {
    // GPS belum valid — tetap kirim tanpa koordinat, HP akan gunakan lokasi sendiri
    Serial.println("[PERTAMINA] GPS tidak valid — kirim tanpa koordinat");
  }

  uint32_t now = millis();
  // Cooldown check
  if((now - nav_pertamina_last_ms) < PERTAMINA_NAV_COOLDOWN &&
     nav_pertamina_last_ms != 0) {
    uint32_t sisa = (PERTAMINA_NAV_COOLDOWN - (now - nav_pertamina_last_ms)) / 1000;
    Serial.printf("[PERTAMINA] Cooldown — tunggu %lu detik lagi\n", sisa);
    return;
  }
  nav_pertamina_last_ms = now;

  // ── Bangun query string dengan koordinat ─────────────────────
  char query[128];
  if(gps_ok && !(lat == 0.0 && lon == 0.0)) {
    // Format: "SPBU Pertamina near -6.123456,106.789012"
    // Chronos / Google Maps akan prioritaskan koordinat ini
    snprintf(query, sizeof(query),
             "%s near %.6f,%.6f",
             PERTAMINA_NAV_QUERY, lat, lon);
  } else {
    snprintf(query, sizeof(query), "%s terdekat", PERTAMINA_NAV_QUERY);
  }

  Serial.printf("[PERTAMINA] Mengirim navigasi: \"%s\"\n", query);

  // ── Kirim via Chronos ─────────────────────────────────────────
  // ChronosESP32 v1.x: gunakan setNavigation jika tersedia
  // Jika API berbeda di versi Anda, sesuaikan method name-nya
  //
  // Opsi A: setNavigation (Chronos ≥ v1.5 dengan Maps integration)
  //   watch.setNavigation(lat, lon, String(query));
  //
  // Opsi B: sendAlert ke HP → user tap untuk buka Maps (kompatibel semua versi)
  //   Format deep-link Android: "geo:lat,lon?q=query"
  //   Atau URL Maps: "https://maps.google.com/?q=query&near=lat,lon"
  char maps_url[256];
  if(gps_ok) {
    snprintf(maps_url, sizeof(maps_url),
             "https://maps.google.com/maps?q=%s&ll=%.6f,%.6f",
             "SPBU+Pertamina", lat, lon);
  } else {
    snprintf(maps_url, sizeof(maps_url),
             "https://maps.google.com/maps?q=%s",
             "SPBU+Pertamina+terdekat");
  }

  // Coba setNavigation() dulu (Chronos fitur navigasi penuh)
  // Jika method tidak ada di versi library Anda, comment baris ini
  // dan aktifkan sendAlert() di bawah
#ifdef CHRONOS_HAS_NAVIGATION
  watch.setNavigation(lat, lon, String(PERTAMINA_NAV_QUERY));
  Serial.println("[PERTAMINA] setNavigation() terkirim");
#else
  // Fallback: kirim sebagai notifikasi dengan URL
  // User perlu tap notif di HP → Maps terbuka
  watch.sendNotification(
    "CarDash Ultra",
    is_critical ? "BBM KRITIS! Navigasi Pertamina:" : "BBM Rendah — Pertamina:",
    String(maps_url)
  );
  Serial.println("[PERTAMINA] sendNotification() fallback terkirim");
#endif

  // ── Update state ──────────────────────────────────────────────
  nav_pertamina_active = true;

  // ── Tampilkan notif popup di dashboard ───────────────────────
  notif_app   = is_critical ? "⛽ BBM KRITIS!" : "⛽ BBM RENDAH";
  notif_title = "Navigasi ke Pertamina";
  notif_msg   = is_critical
                ? "Bensin hampir habis! Mencari SPBU..."
                : "Mencarikan SPBU Pertamina terdekat...";
  notif_show  = true;
  notif_ts    = now;

  Serial.printf("[PERTAMINA] Navigasi %s dikirim ke HP. URL: %s\n",
                is_critical ? "KRITIS" : "RENDAH", maps_url);
}

// ── Dipanggil dari update_fuel_sensor() setiap cycle ─────────────
// Periksa kondisi BBM dan trigger navigasi jika perlu

void check_pertamina_trigger() {
  float pct = fuel.level_pct_smooth;
  bool  low  = (pct < FUEL_LOW_WARN_PCT);
  bool  crit = (pct < FUEL_CRIT_WARN_PCT);
  bool  normal = (pct >= FUEL_LOW_WARN_PCT + FUEL_HYSTERESIS + 2.0f);
                 // +2% hysteresis extra agar tidak toggle bolak-balik

  // Reset semua flag jika BBM kembali normal (isi bensin / error sensor)
  if(normal) {
    if(nav_pertamina_active || nav_pertamina_sent_low || nav_pertamina_sent_crit) {
      Serial.println("[PERTAMINA] BBM normal kembali — reset flag navigasi");
    }
    nav_pertamina_active    = false;
    nav_pertamina_sent_low  = false;
    nav_pertamina_sent_crit = false;
    return;
  }

  // Trigger level KRITIS (prioritas lebih tinggi, trigger ulang)
  if(crit && !nav_pertamina_sent_crit && PERTAMINA_NAV_CRIT_EXTRA) {
    nav_pertamina_sent_crit = true;
    Serial.println("[PERTAMINA] TRIGGER: BBM KRITIS!");
    trigger_pertamina_navigation(true);
    return;
  }

  // Trigger level RENDAH (pertama kali)
  if(low && !nav_pertamina_sent_low) {
    nav_pertamina_sent_low = true;
    Serial.println("[PERTAMINA] TRIGGER: BBM rendah");
    trigger_pertamina_navigation(false);
    return;
  }
}

// ═══════════════════════════════════════════════════════════════════
// BUZZER — Non-blocking State Machine — BARU v2.6
// ═══════════════════════════════════════════════════════════════════
//
// Menggunakan tone() / noTone() ESP32 Arduino untuk PWM buzzer pasif.
// State machine loop() memungkinkan beep pattern tanpa blocking delay().
//
// Pattern yang didukung:
//   buzz_play(count, on_ms, off_ms, freq)
//   Contoh: buzz_play(3, 200, 100, 3000) = 3x beep @ 3kHz, 200ms on, 100ms off

void buzz_play(int count, uint32_t on_ms, uint32_t off_ms, uint16_t freq) {
  if(buz.active) return; // Jangan interrupt beep yang sedang jalan
  buz.active       = true;
  buz.beeps_total  = count;
  buz.beeps_done   = 0;
  buz.tone_on      = true;
  buz.phase_start_ms = millis();
  buz.on_ms        = on_ms;
  buz.off_ms       = off_ms;
  buz.freq         = freq;
  tone(BUZZER_PIN, freq); // Mulai tone pertama
}

// Dipanggil setiap loop() — advance state machine
void buzz_tick() {
  if(!buz.active) return;
  uint32_t now     = millis();
  uint32_t elapsed = now - buz.phase_start_ms;

  if(buz.tone_on) {
    if(elapsed >= buz.on_ms) {
      noTone(BUZZER_PIN);
      buz.tone_on      = false;
      buz.phase_start_ms = now;
      buz.beeps_done++;
      if(buz.beeps_done >= buz.beeps_total) {
        buz.active = false; // Semua beep selesai
      }
    }
  } else {
    if(elapsed >= buz.off_ms) {
      tone(BUZZER_PIN, buz.freq);
      buz.tone_on      = true;
      buz.phase_start_ms = now;
    }
  }
}

// ── Trigger buzzer berdasarkan kondisi sensor ─────────────────────
void buzz_check_alerts() {
  uint32_t now = millis();
  if(buz.active) return; // Jangan tumpuk alert

  // Prioritas: Kritis > Warning (cek kritis dulu)

  // 1. SUHU MESIN KRITIS (>100°C) — 3 beep panjang @ 3kHz
  if(temp_engine > BUZZ_TEMP_CRIT &&
     (now - buzz_last_temp_ms) > BUZZER_COOLDOWN_MS) {
    buzz_last_temp_ms = now;
    buzz_play(3, 400, 150, BUZZER_FREQ_CRIT);
    Serial.printf("[BUZZ] Overheat KRITIS: %.1f°C\n", temp_engine);
    return;
  }

  // 2. SUHU MESIN WARNING (>90°C) — 2 beep @ 2kHz
  if(temp_engine > BUZZ_TEMP_WARN && temp_engine <= BUZZ_TEMP_CRIT &&
     (now - buzz_last_temp_ms) > BUZZER_COOLDOWN_MS) {
    buzz_last_temp_ms = now;
    buzz_play(2, 250, 150, BUZZER_FREQ_WARN);
    Serial.printf("[BUZZ] Overheat WARNING: %.1f°C\n", temp_engine);
    return;
  }

  // 3. BBM KRITIS (<8%) — 5 beep cepat @ 3kHz
  if(fuel.level_pct_smooth < FUEL_CRIT_WARN_PCT &&
     (now - buzz_last_fuel_ms) > BUZZER_COOLDOWN_MS) {
    buzz_last_fuel_ms = now;
    buzz_play(5, 150, 80, BUZZER_FREQ_CRIT);
    Serial.printf("[BUZZ] BBM kritis: %.1f%%\n", fuel.level_pct_smooth);
    return;
  }

  // 4. VOLTASE DROP (<11.8V) — 1 beep panjang @ 2kHz
  if(volt > 0.5f && volt < BUZZ_VOLT_LOW &&
     (now - buzz_last_volt_ms) > BUZZER_COOLDOWN_MS) {
    buzz_last_volt_ms = now;
    buzz_play(1, 800, 0, BUZZER_FREQ_WARN);
    Serial.printf("[BUZZ] Voltase drop: %.2fV\n", volt);
    return;
  }

  // 5. OVERSPEED (>SPEED_WARN_KMH) — 4 beep cepat @ 1.5kHz
  if(speed > BUZZ_SPEED_OVER &&
     (now - buzz_last_speed_ms) > BUZZER_COOLDOWN_MS) {
    buzz_last_speed_ms = now;
    buzz_play(4, 100, 80, BUZZER_FREQ_SPEED);
    Serial.printf("[BUZZ] Overspeed: %d km/h\n", speed);
    return;
  }
}

// ═══════════════════════════════════════════════════════════════════
// SPEED GRAPH — Ring Buffer Update & Waveform LVGL — BARU v2.6
// ═══════════════════════════════════════════════════════════════════
//
// Ring buffer wave_buf[] menyimpan kecepatan tiap detik.
// Setiap frame UI, 27 bar (14 kiri + 13 kanan) di-resize sesuai data.
// Bar terbaru ada di tengah (perbatasan kiri-kanan), scrolling ke kiri.

void wave_push_sample() {
  uint32_t now = millis();
  if(now - wave_last_ms < WAVE_SAMPLE_MS) return;
  wave_last_ms = now;
  // Simpan kecepatan saat ini (cap 255 km/h untuk uint8)
  wave_buf[wave_head] = (uint8_t)constrain(speed, 0, 255);
  wave_head = (wave_head + 1) % WAVE_BUF_SIZE;
}

// Warna bar berdasarkan kecepatan
uint32_t wave_bar_color(int spd) {
  if(spd >= SPEED_WARN_KMH) return C_RED;
  if(spd >= 80)             return C_ORANGE;
  if(spd >= 60)             return C_YELLOW;
  return C_ACCENT;  // cyan untuk kecepatan normal
}

// Update tinggi & warna semua bar waveform
// Dipanggil dari ui_update_all() setiap frame
void wave_update_bars() {
  // Bangun array 27 nilai kecepatan: [0]=terlama, [26]=terbaru
  // Total bar: WAVE_BARS_R(13) kanan + WAVE_BARS_L(14) kiri = 27
  int total = WAVE_BARS_L + WAVE_BARS_R; // 27
  int spd_arr[27];
  for(int i = 0; i < total; i++) {
    // Index ke ring buffer: wave_head-1 = terbaru, wave_head-total = terlama
    int idx = ((wave_head - 1 - i) + WAVE_BUF_SIZE) % WAVE_BUF_SIZE;
    spd_arr[total - 1 - i] = wave_buf[idx]; // balik: [0]=terlama
  }

  // Update bar KIRI: bar[0]=terlama (paling kiri), bar[13]=terbaru (tengah)
  for(int i = 0; i < WAVE_BARS_L; i++) {
    if(!wave_bars_l[i]) continue;
    int spd = spd_arr[i + WAVE_BARS_R]; // ambil dari bagian kanan array
    int h = constrain((spd * WAVE_BAR_MAX_H) / SPEED_WARN_KMH, 1, WAVE_BAR_MAX_H);
    // Posisi Y agar bar rata bawah di tengah BOTTOM_H
    int y = (BOTTOM_H - h) / 2;
    lv_obj_set_size(wave_bars_l[i], 3, h);
    lv_obj_set_pos(wave_bars_l[i], 12 + i * 6, y);
    lv_obj_set_style_bg_color(wave_bars_l[i], lv_color_hex(wave_bar_color(spd)), 0);
  }

  // Update bar KANAN: bar[0]=setelah tengah, bar[12]=terlama (paling kanan)
  for(int i = 0; i < WAVE_BARS_R; i++) {
    if(!wave_bars_r[i]) continue;
    int spd = spd_arr[WAVE_BARS_R - 1 - i]; // terlama di kanan
    int h = constrain((spd * WAVE_BAR_MAX_H) / SPEED_WARN_KMH, 1, WAVE_BAR_MAX_H);
    int y = (BOTTOM_H - h) / 2;
    lv_obj_set_size(wave_bars_r[i], 3, h);
    lv_obj_set_pos(wave_bars_r[i], AV_W - 12 - (WAVE_BARS_R - i) * 6, y);
    lv_obj_set_style_bg_color(wave_bars_r[i], lv_color_hex(wave_bar_color(spd)), 0);
  }
}

// ═══════════════════════════════════════════════════════════════════
// UI UPDATE — dipanggil setiap loop ~16ms
// ═══════════════════════════════════════════════════════════════════

void ui_update_all() {
  uint32_t now = millis();

  // ── STATUS BAR ───────────────────────────────────────────────
  if(lbl_status_gps) {
    // v2.4: tampilkan status Hall Effect WSS, bukan GPS speed
    lv_obj_set_style_text_color(lbl_status_gps,
      wss_ok ? lv_color_hex(C_GREEN) : lv_color_hex(C_ORANGE), 0);
    lv_label_set_text(lbl_status_gps, wss_ok ? "SPD:OK" : "SPD:--");
  }

  if(lbl_status_ble)
    lv_obj_set_style_text_color(lbl_status_ble,
      ble_connected ? lv_color_hex(C_ACCENT) : lv_color_hex(C_RED), 0);
  lv_label_set_text(lbl_status_ble, ble_connected ? "BLE:OK" : "BLE:NO");

  if(lbl_status_clock) {
    char tbuf[8];
    snprintf(tbuf, sizeof(tbuf), "%02d:%02d", clk_h, clk_m);
    lv_label_set_text(lbl_status_clock, tbuf);
  }
  if(lbl_status_wx) {
    char wbuf[16];
    snprintf(wbuf, sizeof(wbuf), "%s %d°C", wx_icon_str(wx_icon_code), wx_temp);
    lv_label_set_text(lbl_status_wx, wbuf);
  }

  // ── SHIFT BAR ────────────────────────────────────────────────
  for(int i = 0; i < 10; i++) {
    if(!pnl_shift[i]) continue;
    int threshold = (i + 1) * (RPM_MAX / 10);
    bool lit = (rpm >= threshold);
    uint32_t col;
    if(!lit)       col = C_DIM;
    else if(i < 4) col = C_GREEN;
    else if(i < 7) col = C_YELLOW;
    else if(i < 9) col = C_ORANGE;
    else           col = C_RED;
    // Flash red zone
    if(lit && i >= 8 && ((now / 150) % 2 == 0)) col = C_DIM;
    lv_obj_set_style_bg_color(pnl_shift[i], lv_color_hex(col), 0);
  }

  // ── RPM ARC ──────────────────────────────────────────────────
  if(arc_rpm) {
    lv_arc_set_value(arc_rpm, rpm);
    lv_color_t rc = color_rpm(rpm);
    lv_obj_set_style_arc_color(arc_rpm, rc, LV_PART_INDICATOR);
  }
  if(lbl_rpm_val) {
    lv_label_set_text_fmt(lbl_rpm_val, "%d", rpm);
    lv_obj_set_style_text_color(lbl_rpm_val, color_rpm(rpm), 0);
  }

  // ── TEMP ─────────────────────────────────────────────────────
  if(lbl_temp_val) {
    lv_label_set_text_fmt(lbl_temp_val, "%.0f° C", temp_engine);
    lv_obj_set_style_text_color(lbl_temp_val, color_temp(temp_engine), 0);
  }

  // ── VOLT ─────────────────────────────────────────────────────
  if(lbl_volt_val) {
    lv_label_set_text_fmt(lbl_volt_val, "%.1f V", volt);
    lv_obj_set_style_text_color(lbl_volt_val, color_volt(volt), 0);
  }

  // ── SPEED ────────────────────────────────────────────────────
  if(lbl_speed) {
    lv_label_set_text_fmt(lbl_speed, "%3d", speed);
    bool warn = (speed > SPEED_WARN_KMH);
    lv_obj_set_style_text_color(lbl_speed,
      warn ? lv_color_hex(C_RED) : lv_color_hex(C_WHITE), 0);
  }

  // ── GEAR (besar center) ───────────────────────────────────────
  if(lbl_gear_big) {
    char gbuf[4];
    if(gear == -1)      strcpy(gbuf, "N");
    else if(gear == 0)  strcpy(gbuf, "R");
    else                snprintf(gbuf, sizeof(gbuf), "%d", gear);
    lv_label_set_text(lbl_gear_big, gbuf);
    lv_obj_set_style_text_color(lbl_gear_big,
      (gear == -1) ? lv_color_hex(C_GRAY) : lv_color_hex(C_ACCENT), 0);
  }

  // ── EFFICIENCY + G INLINE ────────────────────────────────────
  if(lbl_eff) {
    uint32_t ecol = (efficiency_idx > 80) ? C_GREEN :
                    (efficiency_idx > 50) ? C_YELLOW : C_ORANGE;
    uint32_t gcol_v = (fabsf(gforce) > 1.0f) ? C_RED :
                      (fabsf(gforce) > 0.5f) ? C_ORANGE :
                      (fabsf(gforce) > 0.2f) ? C_YELLOW : C_GREEN;
    // LVGL label text — warna inline tidak mudah, update single color
    lv_label_set_text_fmt(lbl_eff, "EFF %.0f%%  ·  G %+.2f",
                          efficiency_idx, gforce);
    lv_obj_set_style_text_color(lbl_eff, lv_color_hex(C_GRAY), 0);
  }

  // ── NAV ──────────────────────────────────────────────────────
  // v2.5: Jika navigasi Pertamina aktif, override label nav
  if(nav_pertamina_active && ble_connected) {
    if(lbl_nav_arrow) lv_label_set_text(lbl_nav_arrow, LV_SYMBOL_GPS);
    // Gunakan warna oranye saat navigasi Pertamina aktif
    if(lbl_nav_arrow)
      lv_obj_set_style_text_color(lbl_nav_arrow, lv_color_hex(C_ORANGE), 0);
    // Jika sudah dapat maneuver dari Chronos, tampilkan normal
    if(nav_maneuver != "SIAP" && nav_distance != "---") {
      if(lbl_nav_arrow) {
        lv_label_set_text(lbl_nav_arrow, nav_arrow_str(nav_maneuver));
        lv_obj_set_style_text_color(lbl_nav_arrow, lv_color_hex(C_ACCENT), 0);
      }
      if(lbl_nav_dist) lv_label_set_text(lbl_nav_dist, nav_distance.c_str());
      if(lbl_nav_road) {
        String road_pertamina = "⛽ " + nav_road;
        lv_label_set_text(lbl_nav_road, road_pertamina.c_str());
      }
    } else {
      // Belum dapat respons Maps — tampilkan status menunggu
      if(lbl_nav_dist) lv_label_set_text(lbl_nav_dist, "Mencari...");
      if(lbl_nav_road) lv_label_set_text(lbl_nav_road, "SPBU Pertamina");
    }
  } else {
    // Navigasi normal
    if(lbl_nav_arrow) {
      lv_label_set_text(lbl_nav_arrow, nav_arrow_str(nav_maneuver));
      lv_obj_set_style_text_color(lbl_nav_arrow, lv_color_hex(C_ACCENT), 0);
    }
    if(lbl_nav_dist) lv_label_set_text(lbl_nav_dist, nav_distance.c_str());
    if(lbl_nav_road) lv_label_set_text(lbl_nav_road, nav_road.c_str());
  }

  // ── LAP ──────────────────────────────────────────────────────
  if(lbl_lap_time && lap_running) {
    lv_label_set_text(lbl_lap_time,
      format_lap_time(now - lap_start_ms).c_str());
  }
  if(lbl_lap_info) {
    uint32_t eng_s = engine_total_s;
    if(engine_running) eng_s += (now - engine_start_ms) / 1000;
    int eh = eng_s / 3600, em = (eng_s % 3600) / 60;
    char info[64];
    snprintf(info, sizeof(info), "Best: %s · #%d · %d:%02dh",
             (lap_best_ms > 0) ? format_lap_time(lap_best_ms).c_str() : "--:--.--",
             lap_count, eh, em);
    lv_label_set_text(lbl_lap_info, info);
  }

  // ── GEAR ROW (kanan) ─────────────────────────────────────────
  // vals: R=0, N=-1, 1..5
  static const int gear_vals[] = {0, -1, 1, 2, 3, 4, 5};
  for(int i = 0; i < 7; i++) {
    if(!gear_boxes[i]) continue;
    bool active = (gear == gear_vals[i]);
    lv_obj_set_style_bg_color(gear_boxes[i],
      active ? lv_color_hex(C_ACCENT) : lv_color_hex(C_PANEL2), 0);
    lv_obj_set_style_border_color(gear_boxes[i],
      active ? lv_color_hex(C_ACCENT) : lv_color_hex(C_BORDER2), 0);
    if(gear_box_lbls[i]) {
      lv_obj_set_style_text_color(gear_box_lbls[i],
        active ? lv_color_hex(C_BG) : lv_color_hex(C_GRAY2), 0);
    }
  }

  // ── FUEL ─────────────────────────────────────────────────────
  float fpct = fuel.level_pct_smooth;
  lv_color_t fc = color_fuel(fpct);
  if(lbl_fuel_pct) {
    lv_label_set_text_fmt(lbl_fuel_pct, "%.0f%%", fpct);
    lv_obj_set_style_text_color(lbl_fuel_pct, fc, 0);
  }
  if(bar_fuel) {
    lv_bar_set_value(bar_fuel, (int)fpct, LV_ANIM_OFF);
    lv_obj_set_style_bg_color(bar_fuel, fc, LV_PART_INDICATOR);
  }
  if(lbl_fuel_sub) {
    char fsub[32];
    snprintf(fsub, sizeof(fsub), "%.1f L  ~%.0f km",
             fuel.level_liter, fuel.range_km);
    lv_label_set_text(lbl_fuel_sub, fsub);
  }
  if(lbl_fuel_cons) {
    char fcons[24];
    snprintf(fcons, sizeof(fcons), "%.1f L/100km", fuel.cons_per_100km);
    lv_label_set_text(lbl_fuel_cons, fcons);
  }

  // ── G-FORCE ──────────────────────────────────────────────────
  if(lbl_gforce) {
    lv_label_set_text_fmt(lbl_gforce, "%+.2fG", gforce);
    lv_obj_set_style_text_color(lbl_gforce, color_gforce(gforce), 0);
  }
  // G-Force bar fill (bipolar)
  if(bar_gf_pos && pnl_gf_bar) {
    int bar_total_w = RIGHT_W - 32;
    int center_x    = bar_total_w / 2;
    float pct = (gforce + 2.0f) / 4.0f;  // -2G..+2G → 0..1
    pct = constrain(pct, 0.0f, 1.0f);
    int fill_w = (int)(fabsf(gforce) / 4.0f * (float)bar_total_w / 2.0f);
    int fill_x = (gforce >= 0) ? center_x : (int)(pct * bar_total_w);
    lv_obj_set_size(bar_gf_pos, fill_w, 4);
    lv_obj_set_pos(bar_gf_pos, fill_x, 0);
    lv_obj_set_style_bg_color(bar_gf_pos, color_gforce(gforce), 0);
  }

  // ── TRIP (v2.4: dari Hall Effect, bukan GPS) ─────────────────
  if(lbl_trip_val) {
    // Tampilkan trip saat ini + total odometer
    char trip_buf[32];
    if(wss_total_km < 10000.0) {
      snprintf(trip_buf, sizeof(trip_buf), "%.1f km  |  ODO %.0f",
               (float)wss_trip_km, (float)wss_total_km);
    } else {
      snprintf(trip_buf, sizeof(trip_buf), "%.1f km", (float)wss_trip_km);
    }
    lv_label_set_text(lbl_trip_val, trip_buf);
  }

  // ── BOTTOM ROAD NAME ─────────────────────────────────────────
  if(lbl_bottom_road) {
    String road_up = nav_road;
    road_up.toUpperCase();
    lv_label_set_text(lbl_bottom_road, road_up.c_str());
  }

  // ── SPEED GRAPH WAVEFORM (v2.6) ──────────────────────────────
  wave_update_bars();

  // ── NOTIFIKASI ───────────────────────────────────────────────
  if(pnl_notif) {
    if(notif_show && (now - notif_ts) < NOTIF_DISPLAY_MS) {
      lv_obj_remove_flag(pnl_notif, LV_OBJ_FLAG_HIDDEN);
      if(lbl_notif_app)   lv_label_set_text(lbl_notif_app,   notif_app.c_str());
      if(lbl_notif_title) lv_label_set_text(lbl_notif_title, notif_title.c_str());
      if(lbl_notif_msg)   lv_label_set_text(lbl_notif_msg,   notif_msg.c_str());
    } else {
      notif_show = false;
      lv_obj_add_flag(pnl_notif, LV_OBJ_FLAG_HIDDEN);
    }
  }
}

// ═══════════════════════════════════════════════════════════════════
// SETUP
// ═══════════════════════════════════════════════════════════════════

void setup() {
  Serial.begin(115200);
  Serial.println("\n[BOOT] CarDash ULTRA v2.6");
  Serial.printf("[BOOT] %s | Chip: %s Rev%d | Flash: %dMB | PSRAM: %dMB\n",
    AV_MODE,
    ESP.getChipModel(), ESP.getChipRevision(),
    (int)(ESP.getFlashChipSize()/1024/1024),
    (int)(ESP.getPsramSize()/1024/1024));

  // ── NVS ──────────────────────────────────────────────────────
  nvs_load();

  // ── FUEL STATE INIT ───────────────────────────────────────────
  memset(&fuel, 0, sizeof(fuel));
  fuel.adc_full       = FUEL_ADC_FULL_DEFAULT;
  fuel.adc_empty      = FUEL_ADC_EMPTY_DEFAULT;
  fuel.cons_per_100km = 10.0f;
  fuel.trip_start_km  = -1.0f;

  // ── GPIO ─────────────────────────────────────────────────────
  // RPM via optocoupler (FALLING = koil memicu)
  pinMode(RPM_PIN, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(RPM_PIN), rpm_isr, FALLING);

  // ── WHEEL SPEED SENSOR — Hall Effect Roda (BARU v2.4) ────────
  // A3144/OH090U: output aktif LOW saat magnet terdeteksi
  // Pull-up eksternal 10kΩ + cap 100nF wajib dipasang di hardware
  pinMode(WSS_PIN, INPUT_PULLUP);
  attachInterrupt(digitalPinToInterrupt(WSS_PIN), wss_isr, FALLING);
  wss_last_calc_ms = millis();
  Serial.printf("[WSS] Hall Effect Speed Sensor aktif — GPIO%d\n", WSS_PIN);
  Serial.printf("[WSS] Keliling ban: %.2fm | Magnet/rev: %d | Interval: %dms\n",
                WHEEL_CIRC_M, MAGNETS_PER_REV, WSS_CALC_INTERVAL);

  // ── BUZZER — Pasif PWM (BARU v2.6) ───────────────────────────
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(BUZZER_PIN, LOW);
  memset(&buz, 0, sizeof(buz));
  // Test beep singkat saat boot (konfirmasi buzzer berfungsi)
  buzz_play(1, 80, 0, 2000);
  delay(100); // Tunggu selesai (hanya di setup, OK untuk blocking)
  noTone(BUZZER_PIN);
  buz.active = false;
  Serial.printf("[BUZZ] Buzzer pasif aktif — GPIO%d\n", BUZZER_PIN);

  // Hall Effect Gear (aktif LOW, open-drain)
  pinMode(GEAR1_PIN, INPUT_PULLUP);
  pinMode(GEAR2_PIN, INPUT_PULLUP);
  pinMode(GEAR3_PIN, INPUT_PULLUP);
  pinMode(GEAR4_PIN, INPUT_PULLUP);
  pinMode(GEAR5_PIN, INPUT_PULLUP);
  pinMode(GEARR_PIN, INPUT_PULLUP);

  // ADC
  analogSetAttenuation(FUEL_ADC_ATTEN);
  analogReadResolution(12);
  pinMode(VOLT_PIN,     INPUT);
  pinMode(FUEL_ADC_PIN, INPUT);

  // ── GPS ──────────────────────────────────────────────────────
  gpsSerial.begin(9600, SERIAL_8N1, GPS_RX, GPS_TX);

  // ── DS18B20 ──────────────────────────────────────────────────
  sensors.begin();

  // ── DISPLAY ──────────────────────────────────────────────────
  gfx.init();
  gfx.setBrightness(BRIGHT_NORMAL);

  // ── LVGL ─────────────────────────────────────────────────────
  lv_init();

  fb1 = (uint16_t*)ps_malloc(AV_W * AV_H * sizeof(uint16_t));
  fb2 = (uint16_t*)ps_malloc(AV_W * AV_H * sizeof(uint16_t));
  if(!fb1 || !fb2) {
    Serial.println("[ERROR] PSRAM FB alloc failed!");
    while(1) delay(1000);
  }

  disp = lv_display_create(AV_W, AV_H);
  lv_display_set_flush_cb(disp, lvgl_flush);
  lv_display_set_buffers(disp, fb1, fb2,
                         AV_W * AV_H * sizeof(uint16_t),
                         LV_DISPLAY_RENDER_MODE_FULL);

  // ── CHRONOS ──────────────────────────────────────────────────
  watch.setConnectionCallback(on_chronos_connect);
  watch.setNotificationCallback(on_chronos_notification);
  watch.setWeatherCallback(on_chronos_weather);
  watch.setNavigationCallback(on_chronos_navigation);
  watch.setRTCCallback(on_chronos_time);
  watch.setButtonCallback(on_chronos_button);
  watch.begin();

  // ── BUILD UI ─────────────────────────────────────────────────
  build_splash_screen();
  lv_screen_load(scr_splash);
  splash_start_ms = millis();

  Serial.println("[BOOT] Setup complete — v2.6 ULTRA | SpeedGraph + Buzzer + Auto-Nav");
}

// ═══════════════════════════════════════════════════════════════════
// LOOP
// ═══════════════════════════════════════════════════════════════════

void loop() {
  uint32_t now = millis();

  // ── SPLASH → MAIN ────────────────────────────────────────────
  if(!splash_done && (now - splash_start_ms) >= SPLASH_DURATION) {
    splash_done = true;
    build_main_screen();
    lv_screen_load_anim(scr_main, LV_SCR_LOAD_ANIM_FADE_IN, 600, 0, true);
  }
  if(!splash_done) {
    int prog = map(now - splash_start_ms, 0, SPLASH_DURATION, 0, 100);
    if(bar_splash_prog) lv_bar_set_value(bar_splash_prog, prog, LV_ANIM_ON);
    lv_task_handler();
    delay(16);
    return;
  }

  // ── BLE / CHRONOS ────────────────────────────────────────────
  watch.loop();

  // ── GPS — Hanya untuk Navigasi & Road Name (v2.4) ────────────
  // Kecepatan TIDAK lagi diambil dari GPS — gunakan Hall Effect WSS
  while(gpsSerial.available()) gps.encode(gpsSerial.read());
  if(gps.location.isValid() && gps.location.age() < 2000) {
    gps_ok = true;
    lat    = gps.location.lat();
    lon    = gps.location.lng();
    // speed dari GPS DIHAPUS — dihandle oleh update_wss_speed()
    update_trip(lat, lon);           // hanya update lat/lon, trip dari WSS
    update_gforce(speed);            // gforce dari WSS speed
  } else {
    gps_ok = false;
  }

  // ── RPM ──────────────────────────────────────────────────────
  static uint32_t      rpm_last_calc   = 0;
  static unsigned long rpm_last_pulses = 0;
  if(now - rpm_last_calc >= RPM_CALC_INTERVAL) {
    unsigned long pulses = rpm_pulses - rpm_last_pulses;
    rpm_last_pulses = rpm_pulses;
    float interval_s = (now - rpm_last_calc) / 1000.0f;
    rpm_last_calc = now;
    if(RPM_NOISE_GATE > 0 && pulses <= (unsigned long)RPM_NOISE_GATE) pulses = 0;
    if(pulses > 0) {
      float new_rpm = (pulses / interval_s / RPM_PULSES_REV) * 60.0f;
      new_rpm = constrain(new_rpm, 0.0f, 15000.0f);
      rpm = (int)(rpm * 0.3f + new_rpm * 0.7f);
      rpm = constrain(rpm, 0, 10000);
    }
  }
  if((now - last_rpm_ms) > RPM_TIMEOUT_MS && rpm > 0) {
    rpm = 0;
  }

  // ── TEMPERATURE ──────────────────────────────────────────────
  static uint32_t temp_last = 0;
  if(now - temp_last > 2000) {
    sensors.requestTemperatures();
    temp_engine = sensors.getTempCByIndex(0);
    temp_last   = now;
  }

  // ── VOLTAGE ──────────────────────────────────────────────────
  static uint32_t volt_last = 0;
  if(now - volt_last > 1000) {
    volt      = read_volt();
    volt_last = now;
  }

  // ── GEAR ─────────────────────────────────────────────────────
  gear = read_gear();

  // ── WHEEL SPEED SENSOR — Hall Effect (BARU v2.4) ─────────────
  // Update kecepatan, trip, dan wss_ok setiap WSS_CALC_INTERVAL ms
  update_wss_speed();
  // G-Force diupdate setelah speed ter-update (pakai speed Hall)
  update_gforce(speed);

  // ── FUEL ─────────────────────────────────────────────────────
  update_fuel_sensor();

  // ── ENGINE HOURS ─────────────────────────────────────────────
  update_engine_hours();

  // ── EFFICIENCY ───────────────────────────────────────────────
  update_efficiency();

  // ── AUTO DIM ─────────────────────────────────────────────────
  update_auto_dim();

  // ── SPEED GRAPH SAMPLE (v2.6) ────────────────────────────────
  wave_push_sample();   // Push 1 sampel/detik ke ring buffer

  // ── BUZZER STATE MACHINE (v2.6) ──────────────────────────────
  buzz_tick();          // Advance buzzer (non-blocking)
  buzz_check_alerts();  // Cek kondisi sensor & trigger beep

  // ── UI UPDATE ────────────────────────────────────────────────
  ui_update_all();

  // ── LVGL TICK & HANDLER ──────────────────────────────────────
  lv_tick_inc(16);
  lv_task_handler();

  delay(16);  // ~60fps
}
